    import pymysql
from flask import *

@app.before_request
def before_request():
    g.db = pymysql.connect(host=host,
                            user=user,

    try:

        g.curs = g.db.cursor()

    12번째줄에 before_request는 디비를 열어준다. pymysql.connect를 하면, 데이터베이스를 리턴해준다. 그리고 g라는 보관소에다가 작업하는 데이터베이스를 저장해준다. db를 열어준다.

    cursor는 쿼리문에 의해 저장된 반환값들을 저장하는 메모리 공간이다.

    @app.route('/room', methods = ['GET'])
def getRoomList():
    g.curs.execute(
        "SELECT room.rid, title, order_date, order_time, place, num_users, stuff_cost, stuff_link "
        "FROM room JOIN stuff on room.rid = stuff.room_id "
        "ORDER BY create_time DESC"
    )
    JA = g.curs.fetchall()

    key = "rooms"
    datas = {}
    datas.setdefault(key, [])

    for i in range(len(JA)):
        data = {}
        data['room_id'] = str(JA[i][0])
        data['title'] = str(JA[i][1])
        data['order_date'] = str(JA[i][2])
        data['order_time'] = str(JA[i][3])
        data['place'] = str(JA[i][4])
        data['num_users'] = str(JA[i][5])
        data['stuff_cost'] = str(JA[i][6])
        data['stuff_link'] = str(JA[i][7])

        datas[key].append(data)

    return datas

    GET 을 통해서 해당 리소스를 조회합니다. select할때, 예를들면, 이미 만들어준 방의 정보를 가져오기만 할때.

    POST 를 통해 해당 URL를 요청하면 리소스를 생성할때.

    PUT 은 리소스를 수정할때. 즉 이미 있는 유저의 정보를 바꾸고 싶을때

    DELETE 는 지워줄때.

    g는 플라스크에 있는 라이브러리 이름이다.

    테이블 설계는 room이라는 큰 테이블 구조에 여러 공통된 정보와 stuff 테이블의 공통된 정보가 아닌 정보를 rid 와 id를 기준으로 합친다. 그리고 함친 테이블에서 필요한 attributes들을 골라서 created_time의 decreasing order를 기준으로 가져온다.

    Tip

    SELECT 는 테이블에 있는 attributes를 가져오는거다.

    FROM 은 어느 테이블에서 가져오는가 이다. 모아안에 테이블이 여러게 있다.

    JOIN 은 다른 테이블에서 같은 아이디 값이 있는 테이블을 합친다는 이야기 이다.

    ORDER BY 는 create_time이라는 attributes를 기준으로 descreasing order로 정렬하겠다 라는 의미이다.

    g.curs.execute 은 sql 쿼리문에서 정보를 가져온다.

    g.curs.fechall() 을 통해서 가져온 정보를 JA에 넣어준다.

    datas.setdefault(key, []) 는 여러개의 이중 jarray를 만들기위해 꼭 필요하다.

    data에 하나의 방의 정보를 넣어준다. 그리고 data를 datas에 rooms: 아래로 넣어준다.

    g.curs.execute(

        "VALUES ('{}', '{}', '{}', '{}', '{}', 0)".format(category, title, creator_email, creator_name, place)
    )
    g.db.commit()

    g.curs.execute("SELECT LAST_INSERT_ID()")
    room_id = g.curs.fetchall()

    g.curs.execute(
        "INSERT INTO stuff(room_id, stuff_link, stuff_cost, order_date, order_time) "
        "VALUES ('{}', '{}', '{}', '{}', '{}')".format(room_id[0][0], stuff_link, stuff_cost, order_date, order_time)
    )
    g.db.commit()

    output = {}
    output['room_id'] = str(room_id[0][0])

    return output

    새로운 방의 정보를 받아야 한다.

    content안에 request.get_json()을 통해서 받아서 넣는다.

    INSERT INTO 는 room이라는 테이블에 넣어준다.

    VALUES 는 값을 정해준다.

    g.db.commit() db를 수정하고 싶을 때는 commit을 해주어야 한다.

    SELECT LAST_INSERT_ID() 는 마지막으로 만들어진 방의 id를 가져온다.

    그 id를 room_id에 넣어준다.

    output['creator_name'] = str(data[7])

    return output
def newUserHasEntered(room_id, email):
    g.curs.execute(
        "INSERT INTO participant(room_id, user_email) "
        "VALUES('{}', '{}')".format(room_id, email)
    )
    g.curs.execute(
        "UPDATE room "
        "SET num_users=num_users+1 "
        "WHERE rid='{}'".format(room_id)
    )
    g.db.commit()

    showChatRoom 함수: 채팅방에 필요한 기능을 불러주는 함수

    isNewUserHasEntered(room_id, user_email) 함수의 기능: 이 유저가 새로운 유저인지 아닌지 판독해주는 것이다.

    newUserHasEntered(room_id, email) 함수의 기능: 새로운 유저를 테이블에 등록해주는 것이다.

    @app.route('/chatroom', methods = ['DELETE'])
def leaveRoom():
    content = request.get_json()
    user_email = content['user_email']
    room_id = content['room_id']

    g.curs.exectue(
        "DELETE FROM participant "
        "WHERE user_email='{}' AND room_id='{}'".format(user_email, room_id)
    )
    g.curs.execute(
        "UPDATE room "
        "SET num_users=num_users-1 "
        "WHERE rid='{}'".format(room_id)
    )
    g.db.commit()

    return "Success"

    leaveRoom() 함수의 역할: 방을 나가는 유저를 지우는 함수.

    DELETE 한 행의 모든 정보를 다 지운다.
